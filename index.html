üìä Data Structure

Each WebSocket message contains a JSON object with the following structure:

{
  "timestamp": 1693938200.123,        // Unix timestamp
  "notes": [...],                     // Detected notes (above threshold)
  "all_notes": [...],                 // All 88 piano notes with values
  "serial": 42,                       // Frame sequence number
  "time_slice": 0,                    // Time slice within frame
  "total_notes_detected": 3,          // Count of detected notes
  "has_notes": true,                  // Boolean flag
  "frame_info": {                     // Threshold settings
    "frame_threshold": 0.3,
    "onset_threshold": 0.5
  }
}

üéµ Note Object Format

Each note in the notes and all_notes arrays:

{
  "note": "C",                        // Note name: A, A#, B, C, C#, D, D#, E, F, F#, G, G#
  "octave": 4,                        // Octave number (0-7 for 88-key piano)
  "midi_note": 60,                    // MIDI note number (21-108)
  "velocity": 0.85,                   // Velocity/strength (0.0-1.0)
  "frame_prob": 0.42,                 // Frame detection probability
  "onset_prob": 0.67,                 // Onset detection probability  
  "offset_prob": 0.12,                // Offset detection probability
  "is_onset": true,                   // Above onset threshold
  "is_frame": false,                  // Above frame threshold
  "strength": 0.85                    // Max of frame_prob and velocity
}

üéº Chord Detection

// Simple chord detection example
function detectChords(noteData) {
    const activeNotes = noteData.notes.map(n => n.note);
    
    // Common chord patterns
    const chordPatterns = {
        'C Major': ['C', 'E', 'G'],
        'A Minor': ['A', 'C', 'E'],
        'F Major': ['F', 'A', 'C'],
        'G Major': ['G', 'B', 'D']
    };
    
    for (const [chordName, pattern] of Object.entries(chordPatterns)) {
        const hasAllNotes = pattern.every(note => 
            activeNotes.includes(note)
        );
        if (hasAllNotes) {
            console.log(`Detected: ${chordName}`);
        }
    }
}

üìà Real-time Analysis

// Track note history for analysis
class NoteAnalyzer {
    constructor() {
        this.noteHistory = [];
        this.maxHistory = 100; // Keep last 100 frames
    }
    
    analyze(noteData) {
        // Add to history
        this.noteHistory.push({
            timestamp: noteData.timestamp,
            notes: noteData.notes.map(n => ({
                note: n.note,
                octave: n.octave,
                strength: n.strength
            }))
        });
        
        // Keep history size manageable
        if (this.noteHistory.length > this.maxHistory) {
            this.noteHistory.shift();
        }
        
        // Calculate average activity
        const recentFrames = this.noteHistory.slice(-10);
        const avgActivity = recentFrames.reduce((sum, frame) => 
            sum + frame.notes.length, 0) / recentFrames.length;
            
        return { avgActivity, recentFrames };
    }
}

üé® Custom Visualization

// Example: Create custom note visualizer
class CustomNoteVisualizer {
    constructor(containerId) {
        this.container = document.getElementById(containerId);
        this.socket = new WebSocket('ws://localhost:8765');
        this.setupWebSocket();
    }
    
    setupWebSocket() {
        this.socket.onmessage = (event) => {
            const noteData = JSON.parse(event.data);
            this.updateVisualization(noteData);
        };
    }
    
    updateVisualization(noteData) {
        // Clear previous visualization
        this.container.innerHTML = '';
        
        // Create elements for detected notes only
        noteData.notes.forEach(note => {
            const noteElement = document.createElement('div');
            noteElement.className = 'custom-note';
            noteElement.textContent = `${note.note}${note.octave}`;
            
            // Scale size by strength
            const scale = 0.5 + (note.strength * 1.5);
            noteElement.style.transform = `scale(${scale})`;
            
            // Color by octave
            noteElement.style.backgroundColor = 
                `hsl(${note.octave * 50}, 70%, 60%)`;
                
            this.container.appendChild(noteElement);
        });
    }
}

// Custom threshold filtering
function applyCustomThresholds(noteData, frameThreshold = 0.3, onsetThreshold = 0.5) {
    return noteData.all_notes.filter(note => {
        const isFrame = note.frame_prob > frameThreshold;
        const isOnset = note.onset_prob > onsetThreshold;
        return isFrame || isOnset;
    });
}

// Velocity-based filtering
function filterByVelocity(noteData, minVelocity = 0.5) {
    return noteData.all_notes.filter(note => 
        note.velocity > minVelocity
    );
}

// Note range filtering (by MIDI numbers)
function filterNoteRange(noteData, minMidi = 21, maxMidi = 108) {
    return noteData.all_notes.filter(note => 
        note.midi_note >= minMidi && note.midi_note <= maxMidi
    );
}

// Temporal filtering (debouncing rapid changes)
class NoteDebouncer {
    constructor(debounceMs = 100) {
        this.debounceMs = debounceMs;
        this.lastNotes = new Map();
    }
    
    filter(noteData) {
        const now = noteData.timestamp * 1000;
        const stableNotes = [];
        
        noteData.notes.forEach(note => {
            const key = `${note.note}-${note.octave}`;
            const lastSeen = this.lastNotes.get(key) || 0;
            
            if (now - lastSeen > this.debounceMs) {
                stableNotes.push(note);
                this.lastNotes.set(key, now);
            }
        });
        
        return stableNotes;
    }
}

üéµ Audio Processing

  • Input Buffer: ~100-200ms (depends on audio device)
  • Frame Size: 2048 samples at 16kHz = 128ms
  • Overlap: 75% overlap for smoother detection

üß† AI Model Processing

  • TensorFlow Lite: ~20-50ms per frame
  • 4 Worker Processes: Parallel processing for real-time
  • Model Type: Unidirectional LSTM (most efficient)

üåê Network & Display

  • WebSocket: ~1-5ms local transmission
  • Browser Rendering: ~16ms (60 FPS)
  • Total Web Latency: ~150-300ms

‚ö° What Affects Latency

  • CPU Speed: Faster CPU = lower AI processing time
  • Audio Device: Professional interfaces have lower buffers
  • Sample Rate: Higher rates need more downsampling
  • System Load: Other apps can add processing delays
